public class Main{
	public static void main(String[] args){
		Scanner sc = new Scanner(system.in);
		int a = sc.nextInt();	
		int b = sc.nextInt();	
		char op = sc.next().charAt(0);
		switch(op){
			case '+':System.out.print(a+b);break;
			case '-':System.out.print(a-b);break;
			case '*':System.out.print(a*b);break;
			case '/':
			if(b!=0)	
				System.out.print(a/b);	
			else
				System.out.print("Error:Divison by Zero);
